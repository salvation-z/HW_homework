git init//创建仓库
git add//添加文件或更新变动（.gitignore保存忽略文件模板）
git status//查看状态（-s简版状态）
git commit//版本提交（-m+"提交信息" -v详细差别 -a一步完成add+commit --amend覆盖提交）
git diff//查看修改
git rm//移除跟踪（跟单独rm区别）
git log//查询提交历史（-p显示差异 -n显示最近n次 --stat显示统计结果 --format指定格式 --graph分支图示 --since只显示指定日期之后）
git reset HEAD filename//撤销暂存区file


git remote//显示远程仓库（-v显示url，add name url添加名为name地址为url的远程仓库）
git fetch [remote-name]//拉取远程仓库内容
git push [remote-name] [branch-name]//上传至远程仓库
git clone [url]//下载整个仓库
